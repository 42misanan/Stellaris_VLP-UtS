planet_event = { 
	#救済措置用イベント、on_actionsからコール
	id = VRPUtS_event.500
	hide_window = yes
	is_triggered_only = yes

	trigger = {
		years_passed > 40 #40年以内には発生しない
        any_country = {
            NOR = { 
                has_global_flag = uni_chance
                has_global_flag = uni_recruited
            }
        }
		owner = {
			NOR = { 
				has_country_flag = uni_abandoned
				has_global_flag = uni_rolled
				is_at_war = yes
				is_ai = yes
			}
		}
		OR = {
			has_building = building_holo_theatres
			has_building = building_hyper_entertainment_forum
			has_building = building_autochthon_monument
			has_building = building_corporate_monument
			has_building = building_noble_estates
			has_building = building_commercial_zone
		}
    }
	immediate = {
		owner = {
			set_global_flag = uni_rolled
		}
		root = {
			planet_event = { id = VRPUtS_event.501 }
		}
	}
}

##### イベント、トリガー元：ランダムなコロニーを選択して発生させる、イベントクラスはplanet_eventに直す

planet_event = {
    #なぜか異常に食糧が減る
    id = VRPUtS_event.501
    title = "VRPUtS_event.501"
    desc = "VRPUtS_event.501.desc"
    picture = GFX_evt_archaeological_dig
    show_sound = event_planetary_riot
    event_chain = VRPUtS_uni_chain

	is_triggered_only = yes

    option = {
		name = VRPUtS_event.501.a #派生 - 泳がしておいて犯人捜しを始める
		trigger = {
            owner = { 
				NOT = {
					is_gestalt = yes
            	}
			}
			any_country = {
				NOR = { 
					has_global_flag = uni_chance
					has_global_flag = uni_recruited
				}
			}
        }
		allow = {
			owner = {
				resource_stockpile_compare = {
                    resource = food
                    value >= 1000
                }
				has_monthly_income = {
					resource = food
					value > 0
				}
			}
		}
		owner = {
			add_resource = {
				food = -1000
			}
			set_global_flag = uni_chance
		}
        begin_event_chain = {
            event_chain = VRPUtS_uni_chain
            target = owner
        }
        hidden_effect = {
            planet_event = { id = VRPUtS_event.511 days = 90 }
        }
	}
    option = {
		name = VRPUtS_event.501.b #放棄 - 補填して終わり
		trigger = { 
			owner = { 
				NOR = {
					has_ethic = ethic_fanatic_authoritarian
					has_ethic = ethic_authoritarian
					is_gestalt = yes
				}
			}
		}
		owner = {
			add_monthly_resource_mult = {
				resource = unity
				value = @tier1unityreward
				min = @tier1unitymin
				max = @tier1unitymax
			}
	        set_country_flag = uni_abandoned
		}
	}

    option = {
		name = VRPUtS_event.501.c #強制放棄 - 軍国主義
		trigger = { 
			owner = { 
				OR = {
					has_ethic = ethic_fanatic_authoritarian
					has_ethic = ethic_authoritarian
				}
			}
		}
		owner = {
			add_monthly_resource_mult = {
				resource = unity
				value = @tier1unityreward
				min = @tier1unitymin
				max = @tier1unitymax
			}
	        set_country_flag = uni_abandoned
		}
	}
	option = {
		name = VRPUtS_event.501.d #強制放棄 - ゲシュタルト意識
		trigger = { 
			owner = {
				is_gestalt = yes 
			}
		}
		owner = {
			add_monthly_resource_mult = {
				resource = unity
				value = @tier1unityreward
				min = @tier1unitymin
				max = @tier1unitymax
			}
	        set_country_flag = uni_abandoned
		}
	}
}

	id = VRPUtS_event.511

	desc = "VRPUtS_event.511.desc"
	picture = GFX_evt_announcement
	show_sound = event_solar_fusion
    event_chain = VRPUtS_uni_chain
	location = root

	is_triggered_only = yes

	option = { #継続
		name = VRPUtS_event.511.a
        allow = {
            owner = {
                resource_stockpile_compare = {
                    resource = trade
                    value >= 1500
                }
                resource_stockpile_compare = {
                    resource = influence
                    value >= 75
                }
            }
        }
        owner = {
            add_resource = {
                trade = -1500
                influence = -75
            }
		}
		random_list = {
			30 = {
				planet_event = { id = VRPUtS_event.511 days = 60 }
			}
			70 = {
				owner = {
					add_event_chain_counter = {
						event_chain = "VRPUtS_uni_chain"
						counter = "VRPUtS_uni_rise"

					}

		}
