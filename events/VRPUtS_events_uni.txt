country_event = {
    #公式には存在しない何かに人々は胸を躍らせているらしい
    id = VRPUtS_event.501
    title = "VRPUtS_event.501"
    desc = "VRPUtS_event.501.desc"
    picture = GFX_evt_alien_hieroglyphics
    show_sound = event_planetary_riot
    event_chain = VRPUtS_uni_chain

    mean_time_to_happen = { days = 2 }

    trigger = {
        any_country = {
            NOR = { 
                has_country_flag = uni_chance
                has_country_flag = uni_recruited
            }
        }
        NOR = { 
            has_country_flag = uni_abandoned
            has_country_flag = uni_chance
            has_country_flag = uni_recruited
            }
        has_situation_flag = divination_occuring
        is_ai = no
    }

    option = {
		name = VRPUtS_event.501.a #派生
		trigger = {
            NOR = {
                has_ethic = ethic_fanatic_militarist
                has_ethic = ethic_militarist
                is_gestalt = yes
            }
            any_country = { NOR = {
                has_country_flag = uni_recruited
                has_country_flag = uni_chance
            }}
        }
        set_country_flag = uni_chance
        begin_event_chain = {
            event_chain = VRPUtS_uni_chain
            target = this
        }
        hidden_effect = {
            country_event = { id = VRPUtS_event.511 days = 90 }
        }
	}
    option = {
		name = VRPUtS_event.501.b #放棄
		trigger = { NOR = {
					has_ethic = ethic_fanatic_militarist
					has_ethic = ethic_militarist
					is_gestalt = yes
				}}
		add_monthly_resource_mult = {
			resource = unity
			value = @tier1unityreward
			min = @tier1unitymin
			max = @tier1unitymax
		}
        set_country_flag = uni_abandoned
	}

    option = {
		name = VRPUtS_event.501.c #強制放棄 - 軍国主義
		trigger = { OR = {
					has_ethic = ethic_fanatic_militarist
					has_ethic = ethic_militarist
				}}
		add_monthly_resource_mult = {
			resource = unity
			value = @tier1unityreward
			min = @tier1unitymin
			max = @tier1unitymax
		}
        set_country_flag = uni_abandoned
	}
	option = {
		name = VRPUtS_event.501.d #強制放棄 - ゲシュタルト意識
		trigger = { 
				is_gestalt = yes 
				}
		add_monthly_resource_mult = {
			resource = unity
			value = @tier1unityreward
			min = @tier1unitymin

		}
        set_country_flag = uni_abandoned

    event_chain = VRPUtS_uni_chain

	is_triggered_only = yes

	option = { #継続
		name = VRPUtS_event.511.a
        allow = {
            root = {
                resource_stockpile_compare = {
                    resource = unity
                    value >= 1000
                }
                resource_stockpile_compare = {
                    resource = influence
                    value >= 50
                }
            }
        }
        root = {
            add_resource = {
                unity = -1000
                influence = -50
            }
            add_event_chain_counter = {
                event_chain = "VRPUtS_uni_chain"
				counter = "VRPUtS_uni_rise"
				amount = 1
			}

            country_event = { id = VRPUtS_event.512 days = 80 }
        }
	}

        end_event_chain = VRPUtS_uni_chain

        set_country_flag = uni_abandoned

}
