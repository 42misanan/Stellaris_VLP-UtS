##### イベント、トリガー元：ランダムなコロニーを選択して発生させる、イベントクラスはplanet_eventに直す
##### on_actions から引っ張ってくる
planet_event = {
    #なぜか異常に食糧が減る
    id = VRPUtS_event.501
    title = "VRPUtS_event.501"
    desc = "VRPUtS_event.501.desc"
    picture = GFX_evt_alien_hieroglyphics
    show_sound = event_planetary_riot
    event_chain = VRPUtS_uni_chain

	pre_triggers = {
		has_owner = yes
		is_occupied_flag = no
	}

    trigger = {
		years_passed > 20 #20年以内には発生しない
        any_country = {
            NOR = { 
                has_country_flag = uni_chance
                has_country_flag = uni_recruited
            }
        }
		owner = {
			NOR = { 
				has_country_flag = uni_abandoned
				has_country_flag = uni_chance
				has_country_flag = uni_recruited
				}
			is_at_war = no
			is_ai = no
		}
		OR = {
			has_building = building_holo_theatres
			has_building = building_hyper_entertainment_forum
			has_building = building_autochthon_monument
			has_building = building_corporate_monument
			has_building = building_noble_estates
			has_building = building_commercial_zone
		}
    }

    option = {
		name = VRPUtS_event.501.a #派生 - 泳がしておいて犯人捜しを始める
		trigger = {
            NOR = {
                has_ethic = ethic_fanatic_authoritarian
                has_ethic = ethic_authoritarian
                is_gestalt = yes
            }
            any_country = { NOR = {
                has_country_flag = uni_recruited
                has_country_flag = uni_chance
            }}
        }
		allow = {
			owner = {
				resource_stockpile_compare = {
                    resource = food
                    value >= 1000
                }
				has_monthly_income = {
					resource = food
					value > 0
				}
			}
		}
		owner = {
			add_resource = {
				food = -1000
			}
		}
        set_country_flag = uni_chance
        begin_event_chain = {
            event_chain = VRPUtS_uni_chain
            target = owner
        }
        hidden_effect = {
            planet_event = { id = VRPUtS_event.511 days = 90 }
        }
	}
    option = {
		name = VRPUtS_event.501.b #放棄 - 補填して終わり
		trigger = { NOR = {
					has_ethic = ethic_fanatic_authoritarian
					has_ethic = ethic_authoritarian
					is_gestalt = yes
				}}
		add_monthly_resource_mult = {
			resource = unity
			value = @tier1unityreward
			min = @tier1unitymin
			max = @tier1unitymax
		}
        set_country_flag = uni_abandoned
	}

    option = {
		name = VRPUtS_event.501.c #強制放棄 - 軍国主義
		trigger = { OR = {
					has_ethic = ethic_fanatic_authoritarian
					has_ethic = ethic_authoritarian
				}}
		add_monthly_resource_mult = {
			resource = unity
			value = @tier1unityreward
			min = @tier1unitymin
			max = @tier1unitymax
		}
        set_country_flag = uni_abandoned
	}
	option = {
		name = VRPUtS_event.501.d #強制放棄 - ゲシュタルト意識
		trigger = { 
				is_gestalt = yes 
				}
		add_monthly_resource_mult = {
			resource = unity
			value = @tier1unityreward
			min = @tier1unitymin
			max = @tier1unitymax
		}
        set_country_flag = uni_abandoned
	}
}
