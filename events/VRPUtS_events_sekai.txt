country_event = {
    #公式には存在しない何かに人々は胸を躍らせているらしい
    id = VRPUtS_event.601
    title = "VRPUtS_event.601"
    desc = "VRPUtS_event.601.desc"
    picture = GFX_evt_alien_hieroglyphics
    show_sound = event_planetary_riot
    event_chain = VRPUtS_sekai_chain

    trigger = {
        any_country = {
            NOR = { 
                has_country_flag = sekai_chance
                has_country_flag = sekai_recruited
            }
        }
        NOR = { 
            has_country_flag = sekai_abandoned
            has_country_flag = sekai_chance
            has_country_flag = sekai_recruited
            }
        has_situation_flag = divination_occuring
        is_ai = no
    }

    option = {
		name = VRPUtS_event.601.a #派生
		trigger = {
            NOR = {
                has_ethic = ethic_fanatic_militarist
                has_ethic = ethic_militarist
                is_gestalt = yes
            }
            any_country = { NOR = {
                has_country_flag = sekai_recruited
                has_country_flag = sekai_chance
            }}
        }
        set_country_flag = sekai_chance
        begin_event_chain = {
            event_chain = VRPUtS_sekai_chain
            target = this
        }
        hidden_effect = {
            country_event = { id = VRPUtS_event.611 days = 90 }
        }
	}
    option = {
		name = VRPUtS_event.601.b #放棄
		trigger = { NOR = {
					has_ethic = ethic_fanatic_militarist
					has_ethic = ethic_militarist
					is_gestalt = yes
				}}
		add_monthly_resource_mult = {
			resource = sekaity
			value = @tier1sekaityreward
			min = @tier1sekaitymin
			max = @tier1sekaitymax
		}
        set_country_flag = sekai_abandoned
	}

    option = {
		name = VRPUtS_event.601.c #強制放棄 - 軍国主義
		trigger = { OR = {
					has_ethic = ethic_fanatic_militarist
					has_ethic = ethic_militarist
				}}
		add_monthly_resource_mult = {
			resource = sekaity
			value = @tier1sekaityreward
			min = @tier1sekaitymin
			max = @tier1sekaitymax
		}
        set_country_flag = sekai_abandoned
	}
	option = {
		name = VRPUtS_event.601.d #強制放棄 - ゲシュタルト意識
		trigger = { 
				is_gestalt = yes 
				}
		add_monthly_resource_mult = {
			resource = sekaity
			value = @tier1sekaityreward
			min = @tier1sekaitymin
			max = @tier1sekaitymax
		}
        set_country_flag = sekai_abandoned
	}
}

country_event = {
    #この運動が激化しないようにしつつ、何かするべきだ

	title = "VRPUtS_event.611"
	desc = "VRPUtS_event.611.desc"
	picture = GFX_evt_alien_planet
	show_sound = event_solar_fusion
    event_chain = VRPUtS_sekai_chain

	is_triggered_only = yes

	option = { #継続
		name = VRPUtS_event.611.a
        allow = {
            root = {
                resource_stockpile_compare = {
                    resource = sekaity
                    value >= 1000
                }
                resource_stockpile_compare = {
                    resource = influence
                    value >= 50
                }
            }
        }
        root = {
            add_resource = {
                sekaity = -1000
                influence = -50
            }
            add_event_chain_counter = {
                event_chain = "VRPUtS_sekai_chain"
				counter = "VRPUtS_sekai_rise"
				amount = 1
			}
        }
        hidden_effect = {
            country_event = { id = VRPUtS_event.612 days = 80 }
        }
	}
    option = { #放棄
        name = VRPUtS_event.611.b
        end_event_chain = VRPUtS_sekai_chain
        remove_country_flag = sekai_chance
        set_country_flag = sekai_abandoned
    }
}
