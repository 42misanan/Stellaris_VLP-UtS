planet_event = { 
	#救済措置用イベント、on_actionsからコール
	id = VRPUtS_event.600
	hide_window = yes
	is_triggered_only = yes

	trigger = {
		years_passed > 40 #40年以内には発生しない
        any_country = {
            NOR = { 
                has_country_flag = sekai_chance
                has_country_flag = sekai_recruited
            }
        }
		owner = {
			NOR = { 
				has_country_flag = sekai_abandoned
				has_country_flag = sekai_rolled
				is_at_war = yes
				is_ai = yes
			}
		}
		OR = {
			has_building = building_holo_theatres
			has_building = building_hyper_entertainment_forum
			has_building = building_autochthon_monument
			has_building = building_corporate_monument
			has_building = building_noble_estates
			has_building = building_commercial_zone
		}
    }
	immediate = {
		owner = {
			set_country_flag = sekai_rolled
			country_event = {
				id = VRPUtS_event.601
				months = 1
			}
		}
	}
}

country_event = {
	id = VRPUtS_event.611
	title = "VRPUtS_event.611"
	desc = "VRPUtS_event.611.desc"
	picture = GFX_evt_city_planet
	show_sound = event_solar_fusion
    event_chain = VRPUtS_sekai_chain

	is_triggered_only = yes

	option = { #継続
		name = VRPUtS_event.611.a
        allow = {
            root = {
                resource_stockpile_compare = {
                    resource = unity
                    value >= 1000
                }
                resource_stockpile_compare = {
                    resource = influence
                    value >= 50
                }
            }
        }
        root = {
            add_resource = {
                unity = -1000
                influence = -50
            }
            add_event_chain_counter = {
                event_chain = "VRPUtS_sekai_chain"
				counter = "VRPUtS_sekai_rise"
				amount = 1
			}
        }
        hidden_effect = {
            country_event = { id = VRPUtS_event.612 months = 2 }
        }
	}
	option = { #ちょっと待ってくれ！
        name = VRPUtS_event.611.b
        custom_tooltip = VRPUtS_event.611.b.tt
		root = {
			country_event = { id = VRPUtS_event.611 months = 12 }
		}
    }
    option = { #放棄
        name = VRPUtS_event.611.c
        end_event_chain = VRPUtS_sekai_chain
        remove_country_flag = sekai_chance
        set_country_flag = sekai_abandoned
    }
}

country_event = {
	id = VRPUtS_event.641
	title = "VRPUtS_event.641"
	desc = "VRPUtS_event.641.desc"
	show_sound = event_airlock
	location = fromfrom
	is_triggered_only = yes

    picture = GFX_leader_recruitment_bg_renowned
	event_window_type = leader_recruit
	notification_event_icon = GFX_message_leader_recruitment_stplus_yukari

	immediate = {
		ethic_leader_creator = {
			SPECIES_CLASS = VRPUtS
			LEADER_PORTRAIT = sekai_04
			CLASS = scientist
			TIER = leader_tier_legendary
			NAME = VRPUtS_NAME_sekai
			LEVEL = 5
			GENDER = indeterminable
			IMMORTAL = yes
			ETHIC = xenophile
			DESC = VRPUtS_sekai_leader_desc
			CATCH_PHRASE = VRPUtS_sekai_leader_catch_phrase
		}
		last_created_leader = {
			add_trait = { trait = subclass_scientist_councilor }
			add_trait = { trait = leader_trait_collaborator_3 }
			add_trait = { trait = trait_ruler_champion_of_the_people_2 }
			add_trait = { trait = trait_ruler_charismatic }
			add_trait = { trait = leader_trait_spark_of_genius }
			add_trait = { trait = leader_trait_adaptable }
			set_age = 16
			set_leader_flag = immune_to_negative_traits
			set_leader_flag = leader_death_events_blocked
			save_global_event_target_as = sekai_leader
		}
	}

	picture_event_data = {
		portrait = event_target:ethic_leader
		room = ethic_spaceship_room
	}

	option = {
		name = VRPUtS_event.641.a
		response_text = VRPUtS_event.641.a.response
		is_dialog_only = yes
	}
	option = {
		name = VRPUtS_event.641.b
		response_text = VRPUtS_event.641.b.response
		is_dialog_only = yes
	}
	option = {
		name = VRPUtS_event.641.c
		response_text = VRPUtS_event.641.c.response
		is_dialog_only = yes
	}
	option = {
		name = VRPUtS_event.641.d
		response_text = VRPUtS_event.641.d.response
		is_dialog_only = yes
	}
	option = {
		name = VRPUtS_event.641.e
		response_text = VRPUtS_event.641.e.response
		is_dialog_only = yes
	}

	option = { # DISMISS
		name = DISMISS
		tag = dismiss_leader
		hidden_effect = {
			set_country_flag = sekai_dismissed
			event_target:sekai_leader = {
				kill_leader = { show_notification = no }
			}
		}
	}
	option = { # HIRE
		name = HIRE
		tag = hire_leader
		hidden_effect = {
			set_country_flag = sekai_recruited
			if = {
				limit = {
					is_variable_set = paragon_memory_vault
				}
				change_variable = {
					which = paragon_memory_vault
					value = 1
				}
			}
			hire_event_leader_effect = {
				LEADER = sekai_leader
			}
		}
	}
