########## 足立レイ - 科学者
country_event = {
	id = VRPUtS_event.131
	title = "VRPUtS_event.131"
	desc = "VRPUtS_event.131.desc"
	show_sound = event_ai_started_01
	location = fromfrom
	is_triggered_only = yes

    picture = GFX_leader_recruitment_bg_renowned
	event_window_type = leader_recruit
	notification_event_icon = GFX_message_leader_recruitment_stplus_yukari

	immediate = {
		ethic_leader_creator = {
			SPECIES_CLASS = VRPUtS
			LEADER_PORTRAIT = rei_02
			CLASS = scientist
			TIER = leader_tier_legendary
			NAME = VRPUtS_NAME_rei
			LEVEL = 5
			GENDER = indeterminable
			IMMORTAL = yes
			ETHIC = gestalt_consciousness
			DESC = VRPUtS_rei_leader_desc
			CATCH_PHRASE = VRPUtS_rei_leader_catch_phrase
		}
		last_created_leader = {
			add_trait = { trait = subclass_scientist_governor }
			add_trait = { trait = leader_trait_ancient_knowledge }
			add_trait = { trait = leader_trait_synthetic }
			add_trait = { trait = leader_trait_architectural_interest }
			add_trait = { trait = leader_trait_adaptable }
			add_trait = { trait = leader_trait_collaborator }
			set_age = 302
			set_leader_flag = immune_to_negative_traits
			set_leader_flag = leader_death_events_blocked
			save_global_event_target_as = rei_leader
		}
	}

	picture_event_data = {
		portrait = event_target:ethic_leader
		room = ethic_spaceship_room
	}

	option = {
		name = VRPUtS_event.131.a
		response_text = VRPUtS_event.131.a.response
		is_dialog_only = yes
	}
	option = {
		name = VRPUtS_event.131.b
		response_text = VRPUtS_event.131.b.response
		is_dialog_only = yes
	}
	option = {
		name = VRPUtS_event.131.c
		response_text = VRPUtS_event.131.c.response
		is_dialog_only = yes
	}
	option = {
		name = VRPUtS_event.131.d
		response_text = VRPUtS_event.131.d.response
		is_dialog_only = yes
	}
	option = {
		name = VRPUtS_event.131.e
		response_text = VRPUtS_event.131.e.response
		is_dialog_only = yes
	}

	option = { # DISMISS
		name = DISMISS
		tag = dismiss_leader
		hidden_effect = {
			set_country_flag = rei_dismissed
			event_target:rei_leader = {
				kill_leader = { show_notification = no }
			}
		}
	}
	option = { # HIRE
		name = HIRE
		tag = hire_leader
		hidden_effect = {
			set_country_flag = rei_recruited
			if = {
				limit = {
					is_variable_set = paragon_memory_vault
				}
				change_variable = {
					which = paragon_memory_vault
					value = 1
				}
			}
			hire_event_leader_effect = {
				LEADER = rei_leader
			}
		}
	}
}
